<script>
    let workersDatabase = JSON.parse(localStorage.getItem('workersDB')) || [];
    let currentPhotoDataUrl = "";

    window.onload = function() {
        const urlParams = new URLSearchParams(window.location.search);
        const dataParam = urlParams.get('data');
        
        if (dataParam) {
            // إذا فتحنا الرابط من الكاميرا، يتم تفعيل العرض العمودي للشهادة مباشرة وبشكل كامل
            document.body.classList.add('view-mode');
            renderCertificateFromQuery(dataParam);
        } else {
            updateDashboardTable();
        }
    };

    function previewImage(event) {
        const file = event.target.files[0];
        if (file) {
            // تقليص حجم الصورة برمجياً لضمان تخزينها بالكامل داخل كود الباركود دون مشاكل
            const reader = new FileReader();
            reader.onload = function(e){
                const img = new Image();
                img.onload = function() {
                    const canvas = document.createElement('canvas');
                    const ctx = canvas.getContext('2d');
                    canvas.width = 120; 
                    canvas.height = 150;
                    ctx.drawImage(img, 0, 0, 120, 150);
                    currentPhotoDataUrl = canvas.toDataURL('image/jpeg', 0.6);
                }
                img.src = e.target.result;
            }
            reader.readAsDataURL(file);
        }
    }

    function saveWorkerData(event) {
        event.preventDefault();
        const worker = {
            id: document.getElementById('workerId').value,
            name: document.getElementById('workerName').value,
            gender: document.getElementById('workerGender').value,
            job: document.getElementById('workerJob').value,
            nation: document.getElementById('workerNation').value,
            expiryDate: document.getElementById('expiryDate').value,
            issuePlace: document.getElementById('issuePlace').value,
            estName: document.getElementById('establishmentName').value,
            photo: currentPhotoDataUrl
        };

        workersDatabase.push(worker);
        localStorage.setItem('workersDB', JSON.stringify(workersDatabase)); 
        updateDashboardTable();
        document.getElementById('workerForm').reset();
        currentPhotoDataUrl = ""; 
        alert('تم الحفظ بنجاح!');
    }

    function updateDashboardTable() {
        const tbody = document.querySelector('#workersTable tbody');
        tbody.innerHTML = "";
        workersDatabase.forEach((worker, index) => {
            tbody.innerHTML += `<tr>
                <td>${worker.id}</td>
                <td><strong>${worker.name}</strong></td>
                <td>${worker.job}</td>
                <td>${worker.estName}</td>
                <td>${worker.expiryDate}</td>
                <td><button class="actions-btn" onclick="downloadPDF(${index})">تحميل الشهادة PDF</button></td>
            </tr>`;
        });
    }

    function generateQRAndFillData(worker, targetQRId) {
        document.getElementById('lblCertId').innerText = "7567-" + worker.id.substring(0,4);
        document.getElementById('lblIssuePlace').innerText = worker.issuePlace;
        document.getElementById('lblExpiry').innerText = worker.expiryDate;
        document.getElementById('lblName').innerText = worker.name;
        document.getElementById('lblNationalId').innerText = worker.id;
        document.getElementById('lblGender').innerText = worker.gender;
        document.getElementById('lblNation').innerText = worker.nation;
        document.getElementById('lblJob').innerText = worker.job;
        document.getElementById('certViewPhoto').src = worker.photo || "";

        const qrContainer = document.getElementById(targetQRId);
        qrContainer.innerHTML = ""; 

        // تحويل كافة بيانات العامل وصورته الشخصية إلى نص مشفر (Base64) لحملها داخل الرابط
        const jsonString = JSON.stringify(worker);
        const encodedData = btoa(unescape(encodeURIComponent(jsonString)));

        const currentSiteUrl = window.location.href.split('?')[0];
        const verificationUrl = `${currentSiteUrl}?data=${encodedData}`;
        
        new QRCode(qrContainer, {
            text: verificationUrl,
            width: 120, height: 120,
            colorDark : "#000000", colorLight : "#ffffff",
            correctLevel : QRCode.CorrectLevel.L
        });
    }

    function downloadPDF(index) {
        const worker = workersDatabase[index];
        document.getElementById('certWrapper').style.display = "block";
        generateQRAndFillData(worker, 'certViewQR');

        setTimeout(() => {
            const element = document.getElementById('certificateTemplate');
            const opt = {
                margin: 0, filename: `شهادة_${worker.name}.pdf`,
                image: { type: 'jpeg', quality: 0.98 },
                html2canvas: { scale: 2, useCORS: true },
                jsPDF: { unit: 'mm', format: 'a4', orientation: 'portrait' }
            };
            html2pdf().set(opt).from(element).save().then(()=>{
                if(!document.body.classList.contains('view-mode')) {
                    document.getElementById('certWrapper').style.display = "none";
                }
            });
        }, 600);
    }

    // فك التشفير وعرض الشهادة عمودياً عند فحص الباركود عبر الهاتف
    function renderCertificateFromQuery(encodedData) {
        try {
            const decodedString = decodeURIComponent(escape(atob(encodedData)));
            const worker = JSON.parse(decodedString);
            if (worker) {
                generateQRAndFillData(worker, 'certViewQR');
            }
        } catch (e) {
            document.getElementById('certificateTemplate').innerHTML = "<h2 style='text-align:center; padding:50px;'>عذراً، رمز الاستجابة السريع يحتوي على بيانات غير صالحة.</h2>";
        }
    }
</script>
